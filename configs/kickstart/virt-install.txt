
###########################
## Install kickstart
###########################

virt-install \
    --connect qemu:///system \
    --name test02 \
    --memory 1024 \
    --vcpus=2 \
    --disk size=8,bus=virtio,cache=writeback \
    --network=network:openstack \
    --sound=none \
    --os-type=linux \
    --os-variant centos7.0 \
    --memballoon virtio \
    --cdrom /var/lib/libvirt/iso/CentOS-7-x86_64-Minimal-1611.iso \
    --
    

# Over network

    --location http://centos.mirror.iweb.ca/7/os/x86_64/ \
    --initrd-inject /home/bl/kickstart.cfg \
    --extra-args "ks=file:/kickstart.cfg"

# Only work as root

    --location /var/lib/libvirt/iso/CentOS-7-x86_64-Minimal-1611.iso \
    --initrd-inject /home/bl/kickstart.cfg \
    --extra-args "ks=file:/kickstart.cfg"
